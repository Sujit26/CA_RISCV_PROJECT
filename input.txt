.data
n : .word 9
.text
main:
	lw x10,n
    jal x1,fact
    beqz x0,Exit
fact:
	addi sp,sp,-8
    sw x1,4(sp)
    sw x10,0(sp)
    li x5,1
    bne x10,x5,Else
    addi sp,sp,8
    jalr x0,x1,0
    Else:
    	addi x10,x10,-1
        jal x1,fact
        lw x11,0(sp)
        mul x10,x10,x11
        lw x1,4(sp)
        addi sp,sp,8
        jalr x0,x1,0
Exit:
